theory Two
imports HOL.Real
begin

text ‚Äπ The two class states that a type's carrier is either infinite, or else it has a finite 
  cardinality of at least 2. It is needed when we depend on having at least two distinguishable
  elements. ‚Ä∫
  
class two =
  assumes card_two: "infinite (UNIV :: 'a set) ‚à® card (UNIV :: 'a set) ‚â• 2"
begin
lemma two_diff: "‚àÉ x y :: 'a. x ‚â† y"
proof -
  obtain A where "finite A" "card A = 2" "A ‚äÜ (UNIV :: 'a set)"
  proof (cases "infinite (UNIV :: 'a set)")
    case True
    with infinite_arbitrarily_large[of "UNIV :: 'a set" 2] that
    show ?thesis by auto
  next
    case False
    with card_two that
    show ?thesis
      by (metis UNIV_bool card_UNIV_bool card_image card_le_inj finite.intros(1) finite_insert finite_subset)
  qed
  thus ?thesis
    by (metis (full_types) One_nat_def Suc_1 UNIV_eq_I card.empty card.insert finite.intros(1) insertCI nat.inject nat.simps(3))
qed
end

  by (intro_classes, auto)

instance nat :: two
  by (intro_classes, auto)

instance int :: two
  by (intro_classes, auto simp add: infinite_UNIV_int)

instance rat :: two
  by (intro_classes, auto simp add: infinite_UNIV_char_0)

instance real :: two
  by (intro_classes, auto simp add: infinite_UNIV_char_0)

  by (intro_classes, auto simp add: infinite_UNIV_listI)

theory Lens_Laws
imports

begin

subsection ‚ÄπLens Signature‚Ä∫
  
text ‚ÄπThis theory introduces the signature of lenses and indentifies the core algebraic hierarchy of lens 
  classes, including laws for well-behaved, very well-behaved, and bijective lenses~\cite{Foster07,Fischer2015,Gibbons17}.‚Ä∫
  
record ('a, 'b) lens =
  lens_get :: "'b ‚áí 'a" ("getƒ±")
  lens_put :: "'b ‚áí 'a ‚áí 'b" ("putƒ±")

type_notation
  lens (infixr "‚üπ" 0)

text ‚Äπ Alternative parameters ordering, inspired by Back and von Wright's refinement 
  calculus~\cite{Back1998}, which similarly uses two functions to characterise updates to variables. ‚Ä∫

abbreviation (input) lens_set :: "('a ‚üπ 'b) ‚áí 'a ‚áí 'b ‚áí 'b" ("lsetƒ±") where
"lens_set ‚â° (Œª X v s. put‚áòX‚áô s v)"

text ‚Äπ
  \begin{figure}
  \begin{center}
    \includegraphics[width=6cm]{figures/Lens}
  \end{center}
  \vspace{-5ex}
  \caption{Visualisation of a simple lens}
  \label{fig:Lens}
  \end{figure}

  A lens $X : \view \lto \src$, for source type $\src$ and view type $\view$, identifies
  $\view$ with a subregion of $\src$~\cite{Foster07,Foster09}, as illustrated in Figure~\ref{fig:Lens}. The arrow denotes
  $X$ and the hatched area denotes the subregion $\view$ it characterises. Transformations on
  $\view$ can be performed without affecting the parts of $\src$ outside the hatched area. The lens
  signature consists of a pair of functions $\lget_X : \src \Rightarrow \view$ that extracts a view
  from a source, and $\lput_X : \src \Rightarrow \view \Rightarrow \src$ that updates a view within
  a given source. ‚Ä∫

named_theorems lens_defs

text ‚Äπ @{text lens_source} gives the set of constructible sources; that is those that can be built
  by putting a value into an arbitrary source. ‚Ä∫

definition lens_source :: "('a ‚üπ 'b) ‚áí 'b set" ("ùíÆƒ±") where
"lens_source X = {s. ‚àÉ v s'. s = put‚áòX‚áô s' v}"

abbreviation some_source :: "('a ‚üπ 'b) ‚áí 'b" ("srcƒ±") where
"some_source X ‚â° (SOME s. s ‚àà ùíÆ‚áòX‚áô)"

definition lens_create :: "('a ‚üπ 'b) ‚áí 'a ‚áí 'b" ("createƒ±") where
[lens_defs]: "create‚áòX‚áô v = put‚áòX‚áô (src‚áòX‚áô) v"

text ‚Äπ Function $\lcreate_X~v$ creates an instance of the source type of $X$ by injecting $v$
  as the view, and leaving the remaining context arbitrary. ‚Ä∫
    
definition lens_update :: "('a ‚üπ 'b) ‚áí ('a ‚áí 'a) ‚áí ('b ‚áí 'b)" ("updateƒ±") where
[lens_defs]: "lens_update X f œÉ = put‚áòX‚áô œÉ (f (get‚áòX‚áô œÉ))"

text ‚Äπ The update function is analogous to the record update function which lifts a function
  on a view type to one on the source type. ‚Ä∫

definition lens_obs_eq :: "('b ‚üπ 'a) ‚áí 'a ‚áí 'a ‚áí bool" (infix "‚âÉƒ±" 50) where
[lens_defs]: "s‚á©1 ‚âÉ‚áòX‚áô s‚á©2 = (s‚á©1 = put‚áòX‚áô s‚á©2 (get‚áòX‚áô s‚á©1))"

  by lens $X$. ‚Ä∫

definition lens_override :: "('b ‚üπ 'a) ‚áí 'a ‚áí 'a ‚áí 'a" (infixl "‚óÉƒ±" 95) where
[lens_defs]: "S‚á©1 ‚óÉ‚áòX‚áô S‚á©2 = put‚áòX‚áô S‚á©1 (get‚áòX‚áô S‚á©2)"

text ‚Äπ Weak lenses are the least constrained class of lenses in our algebraic hierarchy. They
  simply require that the PutGet law~\cite{Foster09,Fischer2015} is satisfied, meaning that
  $\lget$ is the inverse of $\lput$. ‚Ä∫

locale weak_lens =
  fixes x :: "'a ‚üπ 'b" (structure)
  assumes put_get: "get (put œÉ v) = v"

  lemma src_source [simp]: "src ‚àà ùíÆ"

    by (simp add: lens_create_def put_get)

  lemma create_inj: "inj create"
    by (metis create_get injI)

    by (simp add: put_get lens_update_def)

  lemma view_determination: 
    assumes "put œÉ u = put œÅ v"

  lemma put_inj: "inj (put œÉ)"

end

locale mwb_lens = weak_lens +
  assumes put_put: "put (put œÉ v) u = put œÉ u"

  text ‚Äπ Mainly well-behaved lenses give rise to a weakened version of the $get-put$ law, 
    where the source must be within the set of constructible sources. ‚Ä∫

  lemma weak_get_put: "œÉ ‚àà ùíÆ ‚üπ put œÉ (get œÉ) = œÉ"
    by (auto simp add: lens_source_def put_get put_put)

  lemma weak_source_determination:
    assumes "œÉ ‚àà ùíÆ" "œÅ ‚àà ùíÆ" "get œÉ = get œÅ" "put œÉ v = put œÅ v"
    shows "œÉ = œÅ"

   lemma weak_put_eq:
     assumes "œÉ ‚àà ùíÆ" "get œÉ = k" "put œÉ u = put œÅ v"
     shows "put œÅ k = œÉ"

  lemma weak_get_via_put: "s ‚àà ùíÆ ‚üπ get s = (THE v. put s v = s)"
    by (rule sym, auto intro!: the_equality weak_get_put, metis put_get)

subsection ‚ÄπVery Well-behaved Lenses‚Ä∫

text ‚ÄπVery well-behaved lenses combine all three laws, as in the literature~\cite{Foster09,Fischer2015}.

text ‚ÄπThe list lens allows us to view a particular element of a list. In order to show it is mainly
  well-behaved we need to define to additional list functions. The following function adds 
  a number undefined elements to the end of a list.‚Ä∫

"list_pad_out xs k = xs @ replicate (k + 1 - length xs) undefined"

  list if the list isn't long enough first.‚Ä∫

"list_augment xs k v = (list_pad_out xs k)[k := v]"

  the list isn't long enough.‚Ä∫

definition nth' :: "'a list ‚áí nat ‚áí 'a" where
"nth' xs i = (if (length xs > i) then xs ! i else undefined)"

  | read_parent (SOME raw_T) ctxt =
      (case Proof_Context.read_typ_abbrev ctxt raw_T of

  [ Element.Fixes vfixes 
  (* Assume that all lenses are very well-behaved and independent (as in a product space) *)
  , Element.Assumes [((Binding.name "vwbs", [simp])
                     , map (fn vwb => (vwb, [])) vwbs),
                     ((Binding.name "indeps", [simp])
                     , map (fn vwb => (vwb, [])) indeps)]
  ]
  end

  let
    open Lens_Lib;
    open Syntax;
    val imps   = map (fn e => Locale.check thy (e, Position.none)) exts

  (Local_Theory.exit_global o snd o 
  Expression.add_locale (Binding.name ssn) Binding.empty [] (locexs,[]) 
  (mk_statespace_elements vds exts thy)) thy
  end

  end
